### 3.1 Conventions used in SQL expressions

Each `sql_expression` is a **standalone query** returning a single row with a single column `value`. Interaction Brain will wrap it to add dimension grouping when allowed.

Parameters:
- `@start_date DATE = NULL`
- `@end_date DATE = NULL`
- `@as_of_date DATE = CAST(GETUTCDATE() AS DATE)`
- `@fiscal_year_start_month INT = 1`

**Date filtering convention**:
- If windowed: `WHERE txn_date >= @start_date AND txn_date < DATEADD(DAY,1,@end_date)` (or `< @end_date` if end is exclusive)
- If all-time: ignore.

> You should create **source normalization views** (recommended) so these metrics don’t depend on raw column names. For brevity, I’m referencing base tables with assumed columns:
- `dbo.donation_detail(person_id, donation_date, amount, fund, appeal, source_system)`
- `dbo.payment_detail(person_id, payment_date, amount, payment_method, source_system)`
- `dbo.order_detail(person_id, order_date, order_total, order_status, product_name, source_system)`
- `dbo.event_detail(person_id, event_date, event_name, ticket_type, amount)`

### `src/intel/metricExecutor.ts`

```ts
import sql from "mssql";
import { MetricDefinition, MetricRequest, MetricResult } from "./types";

export class MetricExecutor {
  constructor(private pool: sql.ConnectionPool) {}

  async getMetricDefinition(metricKey: string): Promise<MetricDefinition> {
    const r = await this.pool.request()
      .input("metric_key", sql.NVarChar(128), metricKey)
      .query(`
        SELECT metric_key, display_name, description, metric_type, unit, grain, default_time_window, sql_expression
        FROM intel.metric_definition
        WHERE metric_key = @metric_key AND is_active = 1;
      `);

    if (r.recordset.length === 0) throw new Error(`Unknown or inactive metric: ${metricKey}`);
    return r.recordset[0] as MetricDefinition;
  }

  async execute(req: MetricRequest): Promise<MetricResult> {
    const def = await this.getMetricDefinition(req.metricKey);

    const request = this.pool.request();

    // Parameters used by metric SQL
    request.input("start_date", sql.Date, req.startDate ? new Date(req.startDate) : null);
    request.input("end_date", sql.Date, req.endDate ? new Date(req.endDate) : null);
    request.input("as_of_date", sql.Date, req.asOfDate ? new Date(req.asOfDate) : new Date());
    request.input("fiscal_year_start_month", sql.Int, req.fiscalYearStartMonth ?? 1);

    // Safety: enforce single-statement SELECT-only (basic guard)
    const normalized = def.sql_expression.trim().toLowerCase();
    if (!normalized.startsWith("select") && !normalized.startsWith("with")) {
      throw new Error(`Unsafe metric SQL (must start with SELECT/WITH): ${def.metric_key}`);
    }
    if (normalized.includes("insert ") || normalized.includes("update ") || normalized.includes("delete ") || normalized.includes("merge ")) {
      throw new Error(`Unsafe metric SQL (DML not allowed): ${def.metric_key}`);
    }

    const result = await request.query(def.sql_expression);
    const row = result.recordset?.[0];
    const value = row ? (row.value === null || row.value === undefined ? null : Number(row.value)) : null;

    return { metricKey: def.metric_key, value };
  }
}
```

```ts
// src/sqp/compiler.ts
import {
  Breakdown,
  CompiledQuery,
  Filter,
  FilterGroup,
  MetricRequest,
  StructuredQueryPlan,
  TimeWindow,
} from "./types";
import { DIMENSIONS, METRICS, TABLES, DimensionDef, MetricDef } from "./catalog";

export class SQPCompileError extends Error {
  code: string;
  details?: any;
  constructor(code: string, message: string, details?: any) {
    super(message);
    this.code = code;
    this.details = details;
  }
}

type AliasMap = Record<string, string>; // tableName -> alias

function isFilterGroup(x: any): x is FilterGroup {
  return x && typeof x === "object" && (x.op === "and" || x.op === "or") && Array.isArray(x.filters);
}

function assert(condition: any, code: string, message: string, details?: any): asserts condition {
  if (!condition) throw new SQPCompileError(code, message, details);
}

function safeIdent(_s: string): never {
  // We never accept raw identifiers from the plan; only registry IDs.
  throw new SQPCompileError("UNSAFE_IDENTIFIER", "Raw identifiers are not allowed.");
}

function getDimension(id: string): DimensionDef {
  const d = DIMENSIONS[id];
  assert(d, "UNKNOWN_DIMENSION", `Unknown dimension: ${id}`);
  return d;
}

function getMetric(id: string): MetricDef {
  const m = METRICS[id];
  assert(m, "UNKNOWN_METRIC", `Unknown metric: ${id}`);
  return m;
}

function nextParamName(i: number) {
  return `p${i}`;
}

function coerceValue(type: DimensionDef["type"], v: any) {
  if (v === null || v === undefined) return null;
  switch (type) {
    case "number": {
      const n = typeof v === "number" ? v : Number(v);
      assert(Number.isFinite(n), "INVALID_PARAM", `Expected number, got ${v}`);
      return n;
    }
    case "boolean": {
      if (typeof v === "boolean") return v;
      if (v === "true") return true;
      if (v === "false") return false;
      throw new SQPCompileError("INVALID_PARAM", `Expected boolean, got ${v}`);
    }
    case "date":
    case "datetime":
    case "string":
    default:
      return String(v);
  }
}

function dateTruncSql(grain: string, expr: string) {
  // Azure SQL: DATETRUNC exists in SQL Server 2022; many Azure SQL editions support it.
  // If you need compatibility, replace with DATEFROMPARTS/...
  switch (grain) {
    case "day":
    case "week":
    case "month":
    case "quarter":
    case "year":
      return `DATETRUNC(${grain.toUpperCase()}, ${expr})`;
    default:
      throw new SQPCompileError("INVALID_GRAIN", `Unsupported grain: ${grain}`);
  }
}

function resolveTimePreset(preset: string, anchorDate?: string): { start?: string; end?: string } {
  // Keep this deterministic; do not rely on SQL GETDATE() so caching is stable.
  const anchor = anchorDate ? new Date(anchorDate) : new Date();
  const yyyy = anchor.getUTCFullYear();
  const mm = anchor.getUTCMonth(); // 0-11
  const dd = anchor.getUTCDate();

  const iso = (d: Date) => d.toISOString().slice(0, 10);

  const startOfYear = new Date(Date.UTC(yyyy, 0, 1));
  const endOfYear = new Date(Date.UTC(yyyy, 11, 31));

  if (preset === "this_year") return { start: iso(startOfYear), end: iso(endOfYear) };
  if (preset === "last_year") {
    const s = new Date(Date.UTC(yyyy - 1, 0, 1));
    const e = new Date(Date.UTC(yyyy - 1, 11, 31));
    return { start: iso(s), end: iso(e) };
  }
  if (preset === "ytd") return { start: iso(startOfYear), end: iso(new Date(Date.UTC(yyyy, mm, dd))) };
  if (preset === "last_12_months") {
    const end = new Date(Date.UTC(yyyy, mm, dd));
    const start = new Date(Date.UTC(yyyy, mm - 12, dd));
    return { start: iso(start), end: iso(end) };
  }
  if (preset === "last_3_years") {
    const end = new Date(Date.UTC(yyyy, mm, dd));
    const start = new Date(Date.UTC(yyyy - 3, mm, dd));
    return { start: iso(start), end: iso(end) };
  }
  throw new SQPCompileError("UNKNOWN_TIME_PRESET", `Unknown time preset: ${preset}`);
}

function buildJoinGraph(baseTable: string, requiredTables: Set<string>) {
  // Simple strategy: only supports joins defined from base table outward (star-ish).
  const base = TABLES[baseTable];
  assert(base, "UNKNOWN_TABLE", `Unknown base table: ${baseTable}`);

  const joins: Array<{ table: string; alias: string; on: string; type: "inner" | "left" }> = [];
  const aliasMap: AliasMap = { [baseTable]: base.alias };

  for (const t of requiredTables) {
    if (t === baseTable) continue;
    const def = TABLES[t];
    assert(def, "UNKNOWN_TABLE", `Unknown table: ${t}`);
    const join = base.joins.find((j) => j.toTable === t);
    assert(join, "JOIN_NOT_ALLOWED", `No allowed join from ${baseTable} to ${t}`);
    aliasMap[t] = def.alias;
    const on = join.on
      .replaceAll("{base}", aliasMap[baseTable])
      .replaceAll("{to}", aliasMap[t]);
    joins.push({ table: def.name, alias: def.alias, on, type: join.type });
  }

  return { aliasMap, joins };
}

function collectRequiredTables(plan: StructuredQueryPlan): { baseTable: string; required: Set<string> } {
  // Base table: from first metric (must exist) OR subject mapping.
  assert(plan.metrics?.length, "NO_METRICS", "Plan must include at least one metric.");
  const firstMetric = getMetric(plan.metrics[0].id);
  const baseTable = firstMetric.baseTable;

  const required = new Set<string>([baseTable]);

  // Metrics
  for (const mr of plan.metrics) {
    const m = getMetric(mr.id);
    assert(m.baseTable === baseTable, "MIXED_BASE_TABLES", "All metrics must share the same base table for now.", {
      baseTable,
      metric: m.id,
      metricBase: m.baseTable,
    });
    (m.requiresTables ?? []).forEach((t) => required.add(t));
  }

  // Dimensions from breakdowns/filters/time/drilldown
  const addDimTable = (dimId: string) => required.add(getDimension(dimId).table);

  plan.breakdowns?.forEach((b) => addDimTable(b.dimension));
  const walkFilters = (fg?: FilterGroup) => {
    if (!fg) return;
    for (const f of fg.filters) {
      if (isFilterGroup(f)) walkFilters(f);
      else addDimTable(f.field);
    }
  };
  walkFilters(plan.where);
  plan.metrics.forEach((m) => walkFilters(m.where));
  if (plan.time?.dateDimension) addDimTable(plan.time.dateDimension);
  plan.drilldown?.select?.forEach(addDimTable);

  return { baseTable, required };
}

function compileFilter(
  f: Filter,
  aliasMap: AliasMap,
  params: Record<string, any>,
  paramIndexRef: { i: number },
  allowMetricFields = false
): string {
  // Field must be a dimension id (or metric id only if explicitly allowed)
  const dim = DIMENSIONS[f.field];
  const met = METRICS[f.field];

  assert(dim || (allowMetricFields && met), "UNKNOWN_FILTER_FIELD", `Unknown filter field: ${f.field}`);

  if (met) {
    // Metric filters are dangerous; disallow by default.
    throw new SQPCompileError("METRIC_FILTER_NOT_ALLOWED", `Metric filters not allowed: ${f.field}`);
  }

  const col = `${aliasMap[dim.table] ?? safeIdent(dim.table)}.${dim.column}`;

  const op = f.op;
  const ci = !!f.ci;

  const colExpr =
    ci && dim.type === "string" ? `LOWER(${col})` : col;

  const addParam = (val: any) => {
    const name = nextParamName(paramIndexRef.i++);
    params[name] = coerceValue(dim.type, val);
    return `@${name}`;
  };

  switch (op) {
    case "eq":
      assert("value" in f, "INVALID_FILTER", "eq requires value");
      return `${colExpr} = ${addParam(ci && typeof f.value === "string" ? String(f.value).toLowerCase() : f.value)}`;
    case "neq":
      assert("value" in f, "INVALID_FILTER", "neq requires value");
      return `${colExpr} <> ${addParam(ci && typeof f.value === "string" ? String(f.value).toLowerCase() : f.value)}`;
    case "gt":
    case "gte":
    case "lt":
    case "lte": {
      assert("value" in f, `${op.toUpperCase()}_REQUIRES_VALUE`, `${op} requires value`);
      const sqlOp = op === "gt" ? ">" : op === "gte" ? ">=" : op === "lt" ? "<" : "<=";
      return `${colExpr} ${sqlOp} ${addParam(f.value)}`;
    }
    case "between": {
      assert("from" in f && "to" in f, "BETWEEN_REQUIRES_FROM_TO", "between requires from and to");
      const p1 = addParam(f.from);
      const p2 = addParam(f.to);
      return `${colExpr} BETWEEN ${p1} AND ${p2}`;
    }
    case "in":
    case "nin": {
      const list = f.values ?? (Array.isArray((f as any).value) ? (f as any).value : undefined);
      assert(Array.isArray(list) && list.length > 0, "IN_REQUIRES_VALUES", `${op} requires non-empty values`);
      assert(list.length <= 2000, "IN_LIST_TOO_LARGE", "IN list too large (max 2000).");
      const ps = list.map((v) =>
        addParam(ci && typeof v === "string" ? String(v).toLowerCase() : v)
      );
      return `${colExpr} ${op === "in" ? "IN" : "NOT IN"} (${ps.join(", ")})`;
    }
    case "contains":
    case "starts_with":
    case "ends_with": {
      assert(dim.type === "string", "STRING_OP_ON_NONSTRING", `${op} only allowed on string dimensions`);
      assert("value" in f, "LIKE_REQUIRES_VALUE", `${op} requires value`);
      const raw = String(f.value ?? "");
      const escaped = raw.replaceAll("[", "[[]").replaceAll("%", "[%]").replaceAll("_", "[_]");
      const pattern =
        op === "contains" ? `%${escaped}%` : op === "starts_with" ? `${escaped}%` : `%${escaped}`;
      return `${colExpr} LIKE ${addParam(ci ? pattern.toLowerCase() : pattern)} ESCAPE '\\'`;
    }
    case "is_null":
      return `${col} IS NULL`;
    case "is_not_null":
      return `${col} IS NOT NULL`;
    default:
      throw new SQPCompileError("UNKNOWN_COMPARATOR", `Unknown comparator: ${op}`);
  }
}

function compileFilterGroup(
  fg: FilterGroup | undefined,
  aliasMap: AliasMap,
  params: Record<string, any>,
  paramIndexRef: { i: number }
): string | undefined {
  if (!fg) return undefined;
  assert(isFilterGroup(fg), "INVALID_FILTER_GROUP", "Invalid filter group structure");

  const parts: string[] = [];
  for (const node of fg.filters) {
    if (isFilterGroup(node)) {
      const inner = compileFilterGroup(node, aliasMap, params, paramIndexRef);
      if (inner) parts.push(`(${inner})`);
    } else {
      parts.push(compileFilter(node, aliasMap, params, paramIndexRef));
    }
  }
  if (!parts.length) return undefined;
  const op = fg.op.toUpperCase();
  return parts.join(` ${op} `);
}

function compileBreakdowns(
  breakdowns: Breakdown[] | undefined,
  aliasMap: AliasMap
): { select: string[]; groupBy: string[]; allowDims: string[] } {
  if (!breakdowns?.length) return { select: [], groupBy: [], allowDims: [] };

  const select: string[] = [];
  const groupBy: string[] = [];
  const allowDims: string[] = [];

  for (const b of breakdowns) {
    const dim = getDimension(b.dimension);
    const col = `${aliasMap[dim.table]}.${dim.column}`;

    let expr = col;
    if (b.grain) {
      assert(dim.dateLike, "GRAIN_ON_NONDATE", `Grain specified but dimension is not date-like: ${dim.id}`);
      assert(dim.allowedGrains?.includes(b.grain), "GRAIN_NOT_ALLOWED", `Grain ${b.grain} not allowed for ${dim.id}`);
      expr = dateTruncSql(b.grain, col);
    }

    const alias = b.label ?? dim.id.replaceAll(".", "_") + (b.grain ? `_${b.grain}` : "");
    // Alias is generated by us; safe bracket quoting.
    select.push(`${expr} AS [${alias}]`);
    groupBy.push(expr);
    allowDims.push(dim.id);
  }

  return { select, groupBy, allowDims };
}

function applyTimeWindow(
  time: TimeWindow | undefined,
  aliasMap: AliasMap,
  params: Record<string, any>,
  paramIndexRef: { i: number }
): string | undefined {
  if (!time) return undefined;

  const dateDimId = time.dateDimension ?? "donation.date";
  const dim = getDimension(dateDimId);
  assert(dim.dateLike, "TIME_DIM_NOT_DATE", `Time window dimension must be date-like: ${dateDimId}`);

  const col = `${aliasMap[dim.table]}.${dim.column}`;

  let start = time.start;
  let end = time.end;

  if (time.preset) {
    const r = resolveTimePreset(time.preset, time.anchorDate);
    start = start ?? r.start;
    end = end ?? r.end;
  }

  assert(start || end, "INVALID_TIME_WINDOW", "Time window must specify start/end or preset.");

  const clauses: string[] = [];
  const addParam = (val: any) => {
    const name = nextParamName(paramIndexRef.i++);
    params[name] = String(val);
    return `@${name}`;
  };

  if (start) clauses.push(`${col} >= ${addParam(start)}`);
  if (end) clauses.push(`${col} <= ${addParam(end)}`);
  return clauses.join(" AND ");
}

function enforceSafety(plan: StructuredQueryPlan, selectedDims: DimensionDef[]) {
  const allowPII = plan.safety?.allowPII ?? false;
  if (!allowPII) {
    const pii = selectedDims.filter((d) => d.pii);
    assert(pii.length === 0, "PII_NOT_ALLOWED", "PII dimensions requested but not allowed.", {
      pii: pii.map((d) => d.id),
    });
  }

  const maxTop = plan.safety?.maxTop ?? 500;
  const top = plan.limit?.top ?? undefined;
  if (top !== undefined) {
    assert(top > 0 && top <= maxTop, "TOP_LIMIT_EXCEEDED", `TOP must be between 1 and ${maxTop}`);
  }
}

export function compilePlan(plan: StructuredQueryPlan): CompiledQuery {
  // Basic structural validation
  assert(plan?.version === "1.0", "UNSUPPORTED_VERSION", "Only SQP version 1.0 is supported.");
  assert(typeof plan.requestId === "string" && plan.requestId.length > 0, "MISSING_REQUEST_ID", "requestId required.");
  assert(Array.isArray(plan.metrics) && plan.metrics.length > 0, "NO_METRICS", "At least one metric is required.");

  // Validate metrics exist
  const metricDefs = plan.metrics.map((m) => getMetric(m.id));

  // Determine base + required tables
  const { baseTable, required } = collectRequiredTables(plan);

  // Build joins + aliases
  const { aliasMap, joins } = buildJoinGraph(baseTable, required);

  // Compile SELECT: breakdowns + metrics
  const params: Record<string, any> = {};
  const paramIndexRef = { i: 1 };

  const breakdownSql = compileBreakdowns(plan.breakdowns, aliasMap);

  const metricSelect: string[] = [];
  const allowMetrics: string[] = [];
  for (const mr of plan.metrics) {
    const m = getMetric(mr.id);
    allowMetrics.push(m.id);

    // Metric expression: replace {base} with base alias
    let expr = m.expr.replaceAll("{base}", aliasMap[baseTable]);

    // Also allow metric to reference known joined aliases by their table defs
    // (we don't do arbitrary replacement; metric definitions must already reference correct aliases)
    const alias = mr.alias ?? m.id.replaceAll(".", "_");
    metricSelect.push(`${expr} AS [${alias}]`);
  }

  // Safety: gather selected dimensions (breakdowns + drilldown select)
  const selectedDimIds = new Set<string>();
  (plan.breakdowns ?? []).forEach((b) => selectedDimIds.add(b.dimension));
  (plan.drilldown?.select ?? []).forEach((d) => selectedDimIds.add(d));
  const selectedDims = [...selectedDimIds].map(getDimension);
  enforceSafety(plan, selectedDims);

  // WHERE: global + metric-level + time
  const whereParts: string[] = [];

  const globalWhere = compileFilterGroup(plan.where, aliasMap, params, paramIndexRef);
  if (globalWhere) whereParts.push(globalWhere);

  // metric-level where: compile and AND into global WHERE (safe)
  for (const mr of plan.metrics) {
    const mw = compileFilterGroup(mr.where, aliasMap, params, paramIndexRef);
    if (mw) whereParts.push(mw);
  }

  const timeWhere = applyTimeWindow(plan.time, aliasMap, params, paramIndexRef);
  if (timeWhere) whereParts.push(timeWhere);

  const whereSql = whereParts.length ? `WHERE ${whereParts.map((p) => `(${p})`).join(" AND ")}` : "";

  // GROUP BY
  const groupBySql = breakdownSql.groupBy.length ? `GROUP BY ${breakdownSql.groupBy.join(", ")}` : "";

  // ORDER BY: only allow ordering by selected aliases (breakdown aliases or metric aliases)
  const orderByParts: string[] = [];
  const allowedOrderKeys = new Set<string>();

  // breakdown aliases
  for (const b of plan.breakdowns ?? []) {
    const dim = getDimension(b.dimension);
    const alias = b.label ?? dim.id.replaceAll(".", "_") + (b.grain ? `_${b.grain}` : "");
    allowedOrderKeys.add(alias);
  }
  // metric aliases
  for (const mr of plan.metrics) {
    const m = getMetric(mr.id);
    const alias = mr.alias ?? m.id.replaceAll(".", "_");
    allowedOrderKeys.add(alias);
  }

  for (const s of plan.sort ?? []) {
    // sort.by can be metric id, metric alias, dim id, or breakdown alias; normalize to alias
    let sortAlias: string | undefined;

    if (METRICS[s.by]) {
      const m = getMetric(s.by);
      sortAlias = m.id.replaceAll(".", "_");
      // if user used metric id but metric had custom alias, prefer that
      const mr = plan.metrics.find((x) => x.id === s.by);
      if (mr?.alias) sortAlias = mr.alias;
    } else if (DIMENSIONS[s.by]) {
      // only sortable if included as breakdown (otherwise ambiguous)
      const b = (plan.breakdowns ?? []).find((x) => x.dimension === s.by);
      assert(b, "SORT_DIM_NOT_IN_BREAKDOWN", `Cannot sort by dimension not in breakdowns: ${s.by}`);
      const dim = getDimension(s.by);
      sortAlias = b.label ?? dim.id.replaceAll(".", "_") + (b.grain ? `_${b.grain}` : "");
    } else {
      // assume alias
      sortAlias = s.by;
    }

    assert(allowedOrderKeys.has(sortAlias), "SORT_NOT_ALLOWED", `Sort key not allowed: ${s.by}`, {
      allowed: [...allowedOrderKeys],
    });

    const dir = s.dir.toUpperCase() === "ASC" ? "ASC" : "DESC";
    orderByParts.push(`[${sortAlias}] ${dir}`);
  }

  const orderBySql = orderByParts.length ? `ORDER BY ${orderByParts.join(", ")}` : "";

  // TOP/OFFSET
  const top = plan.limit?.top;
  const offset = plan.limit?.offset ?? 0;
  assert(offset >= 0, "INVALID_OFFSET", "offset must be >= 0");

  // Use OFFSET/FETCH only if ORDER BY exists (SQL Server requirement)
  let limitSql = "";
  let topSql = "";
  if (top !== undefined) {
    if (offset > 0) {
      assert(orderByParts.length > 0, "OFFSET_REQUIRES_ORDER", "OFFSET requires ORDER BY.");
      limitSql = `OFFSET ${offset} ROWS FETCH NEXT ${top} ROWS ONLY`;
    } else {
      topSql = `TOP (${top}) `;
    }
  } else if (offset > 0) {
    assert(false, "OFFSET_WITHOUT_TOP", "offset requires top.");
  }

  // FROM + JOINs
  const base = TABLES[baseTable];
  const fromSql = `FROM ${base.name} ${base.alias}`;
  const joinSql = joins
    .map((j) => `${j.type.toUpperCase()} JOIN ${j.table} ${j.alias} ON ${j.on}`)
    .join("\n");

  const selectSql = [...breakdownSql.select, ...metricSelect].join(",\n  ");
  assert(selectSql.length > 0, "EMPTY_SELECT", "Nothing selected.");

  const sql = `
SELECT ${topSql}${selectSql}
${fromSql}
${joinSql ? joinSql : ""}
${whereSql}
${groupBySql}
${orderBySql}
${limitSql}
`.trim();

  return {
    sql,
    params,
    debug: {
      baseTable: base.name,
      joins,
      whereSql: whereSql || undefined,
      groupBySql: groupBySql || undefined,
      orderBySql: orderBySql || undefined,
      limitSql: (topSql || limitSql) ? `${topSql}${limitSql}`.trim() : undefined,
    },
    allowlist: {
      metrics: allowMetrics,
      dimensions: breakdownSql.allowDims,
      tables: [...required],
    },
  };
}
```

- `kind` (`'certified' | 'sqp'`)
- `title`
- `synonyms_json` (JSON array of strings)
- `examples_json` (JSON array of strings)
- `default_params_json` (JSON object)
- `sqp_template_json` (JSON, nullable) — for certified SQP templates
- `enabled` (bit)

### Router code

Create `src/intent/router.ts`:

```ts
// src/intent/router.ts
import { StructuredQueryPlan } from "../sqp/types";
import { compilePlan, SQPCompileError } from "../sqp/compiler";
import { sql } from "drizzle-orm"; // optional; replace with your db layer
import { openai } from "@ai-sdk/openai";
import { generateObject } from "ai";
import { z } from "zod";

export type IntentKind = "certified" | "novel_sqp";

export interface IntentDefinition {
  intent_id: string;
  kind: "certified" | "sqp";
  title: string;
  synonyms: string[];
  examples: string[];
  defaultParams: Record<string, any>;
  sqpTemplate?: StructuredQueryPlan; // for certified
}

export interface IntentClassification {
  intent_id: string | "novel_sqp" | "unknown";
  confidence: number; // 0..1
  params?: Record<string, any>;
  rationale?: string;
}

export interface RouteResult {
  kind: IntentKind;
  intentId?: string;
  confidence: number;
  plan?: StructuredQueryPlan;
  compiled?: { sql: string; params: Record<string, any> };
  errors?: Array<{ code: string; message: string; details?: any }>;
}

const ClassificationSchema = z.object({
  intent_id: z.string(),
  confidence: z.number().min(0).max(1),
  params: z.record(z.any()).optional(),
  rationale: z.string().optional(),
});

function buildClassifierPrompt(userMessage: string, intents: IntentDefinition[]) {
  // Keep prompt deterministic, short, and JSON-only output.
  const intentList = intents.map((i) => ({
    intent_id: i.intent_id,
    title: i.title,
    kind: i.kind,
    synonyms: i.synonyms.slice(0, 12),
    examples: i.examples.slice(0, 12),
    defaultParams: i.defaultParams ?? {},
  }));

  return `
You are Sozo's intent classifier.
Return ONLY valid JSON matching:
{ "intent_id": string, "confidence": number, "params"?: object, "rationale"?: string }

Rules:
- Choose a certified intent_id if the user message clearly matches a certified recipe (synonyms/examples).
- If it does not match any certified intent, return intent_id = "novel_sqp".
- If the message is too vague to route, return intent_id = "unknown" with low confidence.
- Extract params when obvious (e.g., year=2025, top_n=50, campaign="Christmas", channel="stripe").
- Do NOT generate SQL. Do NOT generate an SQP.

User message:
${JSON.stringify(userMessage)}

Available intents:
${JSON.stringify(intentList)}
`.trim();
}

export async function loadIntentDefinitions(db: any): Promise<IntentDefinition[]> {
  // Replace with your Azure SQL access layer.
  // Expected columns: intent_id, kind, title, synonyms_json, examples_json, default_params_json, sqp_template_json, enabled
  const rows = await db.execute(sql`
    SELECT intent_id, kind, title, synonyms_json, examples_json, default_params_json, sqp_template_json
    FROM query_plan
    WHERE enabled = 1
  `);

  return rows.map((r: any) => ({
    intent_id: r.intent_id,
    kind: r.kind,
    title: r.title,
    synonyms: safeJsonArray(r.synonyms_json),
    examples: safeJsonArray(r.examples_json),
    defaultParams: safeJsonObject(r.default_params_json),
    sqpTemplate: r.sqp_template_json ? safeJsonObject(r.sqp_template_json) : undefined,
  }));
}

function safeJsonArray(x: any): string[] {
  try {
    const v = typeof x === "string" ? JSON.parse(x) : x;
    return Array.isArray(v) ? v.map(String) : [];
  } catch {
    return [];
  }
}
function safeJsonObject(x: any): any {
  try {
    return typeof x === "string" ? JSON.parse(x) : x ?? {};
  } catch {
    return {};
  }
}

export async function classifyIntent(userMessage: string, intents: IntentDefinition[]) {
  const prompt = buildClassifierPrompt(userMessage, intents);

  const result = await generateObject({
    model: openai("gpt-4.1-mini"),
    schema: ClassificationSchema,
    prompt,
  });

  // Normalize
  const obj = result.object;
  if (!obj.intent_id) return { intent_id: "unknown", confidence: 0.1 } as IntentClassification;
  return obj as IntentClassification;
}

function applyTemplateParams(template: StructuredQueryPlan, params: Record<string, any>): StructuredQueryPlan {
  // Minimal safe param application: only allow known keys you decide.
  // In production, define per-intent param schema.
  const cloned: StructuredQueryPlan = JSON.parse(JSON.stringify(template));

  if (params.top_n && cloned.limit) cloned.limit.top = Number(params.top_n);
  if (params.year && cloned.time) {
    const y = Number(params.year);
    cloned.time.start = `${y}-01-01`;
    cloned.time.end = `${y}-12-31`;
    delete cloned.time.preset;
  }
  if (params.preset && cloned.time) {
    cloned.time.preset = String(params.preset);
    delete cloned.time.start;
    delete cloned.time.end;
  }
  return cloned;
}

export async function routeMessage(args: {
  db: any;
  userMessage: string;
  buildNovelPlan?: (msg: string) => Promise<StructuredQueryPlan>; // your SQP generator (Part 1/3)
}): Promise<RouteResult> {
  const intents = await loadIntentDefinitions(args.db);
  const classification = await classifyIntent(args.userMessage, intents);

  // Fallback for vague
  if (classification.intent_id === "unknown" || classification.confidence < 0.45) {
    return {
      kind: "novel_sqp",
      confidence: classification.confidence,
      errors: [{ code: "NEED_CLARIFICATION", message: "Message too vague to route; ask a clarifying question." }],
    };
  }

  // Certified path
  const match = intents.find((i) => i.intent_id === classification.intent_id && i.kind === "certified");
  if (match?.sqpTemplate) {
    const plan = applyTemplateParams(match.sqpTemplate, classification.params ?? {});
    try {
      const compiled = compilePlan(plan);
      return {
        kind: "certified",
        intentId: match.intent_id,
        confidence: classification.confidence,
        plan,
        compiled: { sql: compiled.sql, params: compiled.params },
      };
    } catch (e: any) {
      const err =
        e instanceof SQPCompileError
          ? { code: e.code, message: e.message, details: e.details }
          : { code: "COMPILE_FAILED", message: String(e?.message ?? e) };
      return { kind: "certified", intentId: match.intent_id, confidence: classification.confidence, errors: [err] };
    }
  }

  // Novel SQP path
  if (classification.intent_id === "novel_sqp") {
    if (!args.buildNovelPlan) {
      return {
        kind: "novel_sqp",
        confidence: classification.confidence,
        errors: [{ code: "NO_SQP_GENERATOR", message: "No SQP generator configured." }],
      };
    }
    const plan = await args.buildNovelPlan(args.userMessage);
    try {
      const compiled = compilePlan(plan);
      return {
        kind: "novel_sqp",
        confidence: classification.confidence,
        plan,
        compiled: { sql: compiled.sql, params: compiled.params },
      };
    } catch (e: any) {
      const err =
        e instanceof SQPCompileError
          ? { code: e.code, message: e.message, details: e.details }
          : { code: "COMPILE_FAILED", message: String(e?.message ?? e) };
      return { kind: "novel_sqp", confidence: classification.confidence, plan, errors: [err] };
    }
  }

  // If classifier returned a non-certified intent that isn't certified, treat as novel
  return {
    kind: "novel_sqp",
    confidence: classification.confidence,
    errors: [{ code: "INTENT_NOT_EXECUTABLE", message: `Intent not executable: ${classification.intent_id}` }],
  };
}
```

**How synonyms/examples feed classification:** `loadIntentDefinitions()` pulls them from `query_plan` and injects into the classifier prompt as the “Available intents” list.

```ts
// src/evidence/buildEvidence.ts
import { EvidencePack, StructuredQueryPlan } from "../sqp/types";
import { CompiledQuery } from "../sqp/types";

export async function executeCompiledQuery(db: any, compiled: CompiledQuery) {
  // Replace with your Azure SQL client that supports named params.
  // Must pass params separately; never string interpolate.
  const result = await db.query(compiled.sql, compiled.params);
  return result; // expect { rows: any[], columns?: ... }
}

export function buildEvidencePack(args: {
  plan: StructuredQueryPlan;
  compiled: CompiledQuery;
  result: { rows: any[]; columns?: Array<{ name: string; type: string }> };
  warnings?: string[];
}): EvidencePack {
  const rows = args.result.rows ?? [];
  const columns =
    args.result.columns ??
    (rows[0] ? Object.keys(rows[0]).map((k) => ({ name: k, type: typeof (rows[0] as any)[k] })) : []);

  return {
    requestId: args.plan.requestId,
    generatedAt: new Date().toISOString(),
    plan: args.plan,
    compiled: args.compiled,
    facts: {
      rowCount: rows.length,
      columns,
      sampleRows: rows.slice(0, 20),
      summaries: {},
    },
    provenance: {
      tables: args.compiled.allowlist.tables,
      metrics: args.compiled.allowlist.metrics,
      dimensions: args.compiled.allowlist.dimensions,
      timeWindow: args.plan.time
        ? {
            start: args.plan.time.start,
            end: args.plan.time.end,
            preset: args.plan.time.preset,
            dateDimension: args.plan.time.dateDimension,
          }
        : undefined,
      warnings: args.warnings ?? [],
    },
  };
}
```

### 1.1 `intel.vw_donor_health`
RFM + engagement + trajectory + risk level **per person**.

```sql
CREATE OR ALTER VIEW intel.vw_donor_health
AS
WITH gifts AS (
    SELECT
        g.person_id,
        CAST(g.gift_date AS date) AS gift_date,
        CAST(g.amount AS decimal(18,2)) AS amount
    FROM dbo.gift g
    WHERE g.status = 'posted'
),
agg AS (
    SELECT
        person_id,
        MAX(gift_date) AS last_gift_date,
        COUNT(*) AS gift_count_lifetime,
        SUM(amount) AS gift_amount_lifetime,
        SUM(CASE WHEN gift_date >= DATEADD(day, -365, CAST(GETDATE() AS date)) THEN amount ELSE 0 END) AS gift_amount_365,
        COUNT(CASE WHEN gift_date >= DATEADD(day, -365, CAST(GETDATE() AS date)) THEN 1 END) AS gift_count_365,
        SUM(CASE WHEN gift_date >= DATEADD(day, -90, CAST(GETDATE() AS date)) THEN amount ELSE 0 END) AS gift_amount_90,
        COUNT(CASE WHEN gift_date >= DATEADD(day, -90, CAST(GETDATE() AS date)) THEN 1 END) AS gift_count_90
    FROM gifts
    GROUP BY person_id
),
eng AS (
    SELECT
        pt.person_id,
        COUNT(*) AS tag_events_365
    FROM dbo.person_tag pt
    WHERE pt.created_at >= DATEADD(day, -365, GETDATE())
    GROUP BY pt.person_id
),
trend AS (
    SELECT
        g.person_id,
        SUM(CASE WHEN g.gift_date >= DATEADD(day, -365, CAST(GETDATE() AS date))
                  AND g.gift_date <  DATEADD(day, -182, CAST(GETDATE() AS date))
                 THEN g.amount ELSE 0 END) AS amt_prev_6mo,
        SUM(CASE WHEN g.gift_date >= DATEADD(day, -182, CAST(GETDATE() AS date))
                 THEN g.amount ELSE 0 END) AS amt_last_6mo
    FROM gifts g
    GROUP BY g.person_id
),
rfm AS (
    SELECT
        a.person_id,
        DATEDIFF(day, a.last_gift_date, CAST(GETDATE() AS date)) AS recency_days,
        a.gift_count_365 AS frequency_365,
        a.gift_amount_365 AS monetary_365,
        -- R score: lower recency is better
        CASE
            WHEN a.last_gift_date IS NULL THEN 1
            WHEN DATEDIFF(day, a.last_gift_date, CAST(GETDATE() AS date)) <= 30 THEN 5
            WHEN DATEDIFF(day, a.last_gift_date, CAST(GETDATE() AS date)) <= 90 THEN 4
            WHEN DATEDIFF(day, a.last_gift_date, CAST(GETDATE() AS date)) <= 180 THEN 3
            WHEN DATEDIFF(day, a.last_gift_date, CAST(GETDATE() AS date)) <= 365 THEN 2
            ELSE 1
        END AS r_score,
        -- F score
        CASE
            WHEN a.gift_count_365 >= 12 THEN 5
            WHEN a.gift_count_365 >= 6 THEN 4
            WHEN a.gift_count_365 >= 3 THEN 3
            WHEN a.gift_count_365 >= 1 THEN 2
            ELSE 1
        END AS f_score,
        -- M score
        CASE
            WHEN a.gift_amount_365 >= 5000 THEN 5
            WHEN a.gift_amount_365 >= 1000 THEN 4
            WHEN a.gift_amount_365 >= 250 THEN 3
            WHEN a.gift_amount_365 > 0 THEN 2
            ELSE 1
        END AS m_score
    FROM agg a
)
SELECT
    p.person_id,
    a.last_gift_date,
    r.recency_days,
    r.frequency_365,
    r.monetary_365,
    r.r_score,
    r.f_score,
    r.m_score,
    (r.r_score + r.f_score + r.m_score) AS rfm_score,
    ISNULL(e.tag_events_365, 0) AS engagement_depth_365,
    t.amt_last_6mo,
    t.amt_prev_6mo,
    CASE
        WHEN ISNULL(t.amt_prev_6mo,0) = 0 AND ISNULL(t.amt_last_6mo,0) > 0 THEN 1.0
        WHEN ISNULL(t.amt_prev_6mo,0) = 0 THEN 0.0
        ELSE (t.amt_last_6mo - t.amt_prev_6mo) / NULLIF(t.amt_prev_6mo,0)
    END AS growth_trajectory_6mo_pct,
    CASE
        WHEN a.last_gift_date IS NULL THEN 'lost'
        WHEN DATEDIFF(day, a.last_gift_date, CAST(GETDATE() AS date)) > 365 THEN 'lost'
        WHEN DATEDIFF(day, a.last_gift_date, CAST(GETDATE() AS date)) > 180 THEN 'lapsed'
        WHEN DATEDIFF(day, a.last_gift_date, CAST(GETDATE() AS date)) > 90 THEN 'cooling'
        ELSE 'active'
    END AS risk_level
FROM dbo.person p
LEFT JOIN agg a ON a.person_id = p.person_id
LEFT JOIN rfm r ON r.person_id = p.person_id
LEFT JOIN eng e ON e.person_id = p.person_id
LEFT JOIN trend t ON t.person_id = p.person_id;
GO
```

### 1.2 `intel.vw_person_value`
Ministry lifetime value combining giving + commerce + events + subscriptions.

```sql
CREATE OR ALTER VIEW intel.vw_person_value
AS
WITH giving AS (
    SELECT person_id, SUM(CAST(amount AS decimal(18,2))) AS giving_ltv
    FROM dbo.gift
    WHERE status = 'posted'
    GROUP BY person_id
),
commerce AS (
    SELECT person_id, SUM(CAST(total AS decimal(18,2))) AS commerce_ltv
    FROM dbo.order_header
    WHERE status IN ('paid','fulfilled')
    GROUP BY person_id
),
events AS (
    SELECT person_id, SUM(CAST(price AS decimal(18,2))) AS event_ltv
    FROM dbo.event_ticket
    WHERE status IN ('paid','attended')
    GROUP BY person_id
),
subs AS (
    SELECT
        person_id,
        SUM(
            CAST(monthly_amount AS decimal(18,2)) *
            CAST(
                DATEDIFF(month, start_date, ISNULL(end_date, CAST(GETDATE() AS date))) + 1
            AS int)
        ) AS subscription_ltv
    FROM dbo.subscription
    WHERE status IN ('active','canceled','ended')
    GROUP BY person_id
)
SELECT
    p.person_id,
    ISNULL(g.giving_ltv, 0) AS giving_ltv,
    ISNULL(c.commerce_ltv, 0) AS commerce_ltv,
    ISNULL(e.event_ltv, 0) AS event_ltv,
    ISNULL(s.subscription_ltv, 0) AS subscription_ltv,
    ISNULL(g.giving_ltv, 0) + ISNULL(c.commerce_ltv, 0) + ISNULL(e.event_ltv, 0) + ISNULL(s.subscription_ltv, 0) AS total_ltv
FROM dbo.person p
LEFT JOIN giving g ON g.person_id = p.person_id
LEFT JOIN commerce c ON c.person_id = p.person_id
LEFT JOIN events e ON e.person_id = p.person_id
LEFT JOIN subs s ON s.person_id = p.person_id;
GO
```

### 1.5 `intel.vw_event_giving_lift`
Giving before vs after event attendance per person (30/90-day windows).

```sql
CREATE OR ALTER VIEW intel.vw_event_giving_lift
AS
WITH attended AS (
    SELECT
        t.person_id,
        MIN(CAST(t.event_date AS date)) AS first_event_date,
        MAX(CAST(t.event_date AS date)) AS last_event_date,
        COUNT(*) AS events_attended
    FROM dbo.event_ticket t
    WHERE t.status IN ('attended','paid')
    GROUP BY t.person_id
),
giving AS (
    SELECT person_id, CAST(gift_date AS date) AS gift_date, CAST(amount AS decimal(18,2)) AS amount
    FROM dbo.gift
    WHERE status='posted'
)
SELECT
    a.person_id,
    a.first_event_date,
    a.last_event_date,
    a.events_attended,
    SUM(CASE WHEN g.gift_date >= DATEADD(day,-90,a.first_event_date) AND g.gift_date < a.first_event_date THEN g.amount ELSE 0 END) AS giving_90d_before,
    SUM(CASE WHEN g.gift_date >= a.first_event_date AND g.gift_date < DATEADD(day,90,a.first_event_date) THEN g.amount ELSE 0 END) AS giving_90d_after,
    SUM(CASE WHEN g.gift_date >= DATEADD(day,-30,a.first_event_date) AND g.gift_date < a.first_event_date THEN g.amount ELSE 0 END) AS giving_30d_before,
    SUM(CASE WHEN g.gift_date >= a.first_event_date AND g.gift_date < DATEADD(day,30,a.first_event_date) THEN g.amount ELSE 0 END) AS giving_30d_after,
    (SUM(CASE WHEN g.gift_date >= a.first_event_date AND g.gift_date < DATEADD(day,90,a.first_event_date) THEN g.amount ELSE 0 END)
     -
     SUM(CASE WHEN g.gift_date >= DATEADD(day,-90,a.first_event_date) AND g.gift_date < a.first_event_date THEN g.amount ELSE 0 END)
    ) AS giving_lift_90d
FROM attended a
LEFT JOIN giving g ON g.person_id = a.person_id
GROUP BY a.person_id, a.first_event_date, a.last_event_date, a.events_attended;
GO
```

### 1.9 `intel.vw_concentration_risk`
Top N donors as % of total (parameterless view: uses N=10/25/50 columns).

```sql
CREATE OR ALTER VIEW intel.vw_concentration_risk
AS
WITH y AS (
    SELECT
        person_id,
        SUM(CAST(amount AS decimal(18,2))) AS giving_ytd
    FROM dbo.gift
    WHERE status='posted'
      AND gift_date >= DATEFROMPARTS(YEAR(GETDATE()),1,1)
      AND gift_date < DATEADD(day,1,CAST(GETDATE() AS date))
    GROUP BY person_id
),
tot AS (
    SELECT SUM(giving_ytd) AS total_ytd FROM y
),
ranked AS (
    SELECT
        person_id,
        giving_ytd,
        ROW_NUMBER() OVER (ORDER BY giving_ytd DESC) AS rn
    FROM y
)
SELECT
    t.total_ytd,
    SUM(CASE WHEN rn <= 10 THEN giving_ytd ELSE 0 END) AS top10_ytd,
    SUM(CASE WHEN rn <= 25 THEN giving_ytd ELSE 0 END) AS top25_ytd,
    SUM(CASE WHEN rn <= 50 THEN giving_ytd ELSE 0 END) AS top50_ytd,
    CASE WHEN t.total_ytd=0 THEN 0 ELSE SUM(CASE WHEN rn <= 10 THEN giving_ytd ELSE 0 END)/t.total_ytd END AS top10_pct,
    CASE WHEN t.total_ytd=0 THEN 0 ELSE SUM(CASE WHEN rn <= 25 THEN giving_ytd ELSE 0 END)/t.total_ytd END AS top25_pct,
    CASE WHEN t.total_ytd=0 THEN 0 ELSE SUM(CASE WHEN rn <= 50 THEN giving_ytd ELSE 0 END)/t.total_ytd END AS top50_pct
FROM ranked
CROSS JOIN tot t;
GO
```

## 2) Nightly Job: Metric Snapshotter (TypeScript)

### 2.1 Metric definition contract
We’ll support metrics defined as SQL expressions with optional segment queries.

**Recommended fields** in `intel.metric_definition` (if you already have it, map accordingly):
- `metric_definition_id uniqueidentifier`
- `metric_key nvarchar(200)` unique
- `name nvarchar(200)`
- `description nvarchar(1000)`
- `certified bit`
- `value_sql nvarchar(max)` — returns a single row with column `value` (decimal)
- `segments_json nvarchar(max)` — JSON array of segments: `{ key, name, whereSql }`
- `unit nvarchar(50)` e.g. `usd`, `count`, `pct`
- `format nvarchar(50)` e.g. `currency`, `number`, `percent`

```ts
  const req = db.request();
  if (params) {
    for (const [k, v] of Object.entries(params)) req.input(k, v as any);

  return req.query<T>(text);
}
```

### 2.3 Snapshotter implementation

`src/jobs/metricSnapshotter.ts`
```ts
import { getDb, query } from "@/lib/db";
import type { MetricDefinition, SegmentDef } from "@/lib/intel/types";

type SnapshotRow = {
  metric_definition_id: string;
  metric_key: string;
  segment_key: string | null;
  segment_name: string | null;
  as_of_date: string; // yyyy-mm-dd
  value: number;
  wow_delta: number | null;
  mom_delta: number | null;
  yoy_delta: number | null;
};

function asDateOnly(d = new Date()): string {
  const yyyy = d.getUTCFullYear();
  const mm = String(d.getUTCMonth() + 1).padStart(2, "0");
  const dd = String(d.getUTCDate()).padStart(2, "0");
  return `${yyyy}-${mm}-${dd}`;
}

function safeJsonParse<T>(s: string | null): T | null {
  if (!s) return null;
  try {
    return JSON.parse(s) as T;
  } catch {
    return null;
  }
}

/**
 * Wraps a metric's value_sql and applies a segment WHERE clause if provided.
 * Contract: metric value_sql must be a SELECT that returns a single row with column [value].
 * To segment, we inject the whereSql into a placeholder token: {{where}}
 *
 * Example value_sql:
 *   SELECT CAST(SUM(amount) AS decimal(18,2)) AS value
 *   FROM dbo.gift
 *   WHERE status='posted' {{where}}
 */
function buildSegmentedSql(valueSql: string, whereSql?: string): string {
  const token = "{{where}}";
  const clause = whereSql ? ` AND (${whereSql})` : "";
  if (valueSql.includes(token)) return valueSql.replaceAll(token, clause);

  // If no token, we can still attempt to append by wrapping:
  return `
    SELECT value
    FROM (
      ${valueSql}
    ) base
  `;
}

async function computeValue(db: any, sqlText: string): Promise<number> {
  const res = await query<{ value: any }>(db, sqlText);
  const row = res.recordset[0];
  const v = row?.value;
  const num = v === null || v === undefined ? 0 : Number(v);
  if (Number.isNaN(num)) throw new Error(`Metric returned non-numeric value: ${v}`);
  return num;
}

async function getPriorValue(
  db: any,
  metricKey: string,
  segmentKey: string | null,
  asOfDate: string,
  daysBack: number
): Promise<number | null> {
  const res = await query<{ value: number }>(
    db,
    `
    SELECT TOP 1 value
    FROM intel.metric_snapshot
    WHERE metric_key = @metricKey
      AND ((@segmentKey IS NULL AND segment_key IS NULL) OR segment_key = @segmentKey)
      AND as_of_date = DATEADD(day, -@daysBack, @asOfDate)
    `,
    { metricKey, segmentKey, asOfDate, daysBack }
  );
  return res.recordset[0]?.value ?? null;
}

function delta(current: number, prior: number | null): number | null {
  if (prior === null) return null;
  return current - prior;
}
```

## 3) Nightly Job: Anomaly Detector

### 3.1 Statistical algorithm (robust seasonal z-score)

For each metric (and segment), for “today”:
1. Gather history for the last 2–3 years (or as available).
2. Build a **seasonal peer group**: same **month** (and optionally week-of-year) across prior years.
   - This captures December giving spikes naturally.
3. Compute **median** and **MAD** (median absolute deviation) of the peer group.
4. Robust z-score:

\[
z = 0.6745 \cdot \frac{x - \text{median}}{\text{MAD} + \epsilon}
\]

Flag anomaly if `|z| >= threshold` (e.g. 3.5). Severity can be `min(1, |z|/8)`.

### 3.2 Implementation

`src/jobs/anomalyDetector.ts`
```ts
import { getDb, query } from "@/lib/db";

type Snap = {
  metric_key: string;
  segment_key: string | null;
  as_of_date: string; // date
  value: number;
};

function median(arr: number[]): number {
  if (arr.length === 0) return 0;
  const a = [...arr].sort((x, y) => x - y);
  const mid = Math.floor(a.length / 2);
  return a.length % 2 ? a[mid] : (a[mid - 1] + a[mid]) / 2;
}

function mad(arr: number[], med: number): number {
  const dev = arr.map((x) => Math.abs(x - med));
  return median(dev);
}

function robustZ(x: number, med: number, madVal: number): number {
  const eps = 1e-9;
  return 0.6745 * (x - med) / (madVal + eps);
}

function severityFromZ(z: number): number {
  const az = Math.abs(z);
  return Math.max(0, Math.min(1, az / 8));
}

export async function runAnomalyDetector(now = new Date()): Promise<void> {
  const db = await getDb();

  // Today snapshot rows
  const todayRes = await query<Snap>(
    db,
    `
    SELECT metric_key, segment_key, CONVERT(varchar(10), as_of_date, 23) AS as_of_date, CAST(value AS float) AS value
    FROM intel.metric_snapshot
    WHERE as_of_date = CAST(GETDATE() AS date)
    `
  );
  const todays = todayRes.recordset;

  for (const cur of todays) {
    // Seasonal peer group: same month in prior 3 years, excluding today
    const histRes = await query<{ value: number }>(
      db,
      `
      SELECT CAST(value AS float) AS value
      FROM intel.metric_snapshot
      WHERE metric_key = @metricKey
        AND ((@segmentKey IS NULL AND segment_key IS NULL) OR segment_key = @segmentKey)
        AND as_of_date < CAST(GETDATE() AS date)
        AND as_of_date >= DATEADD(year, -3, CAST(GETDATE() AS date))
        AND MONTH(as_of_date) = MONTH(CAST(GETDATE() AS date))
      `,
      { metricKey: cur.metric_key, segmentKey: cur.segment_key }
    );

    const values = histRes.recordset.map((r) => Number(r.value)).filter((v) => Number.isFinite(v));
    if (values.length < 10) continue; // not enough history

    const med = median(values);
    const madVal = mad(values, med);
    const z = robustZ(cur.value, med, madVal);

    const threshold = 3.5;
    if (Math.abs(z) < threshold) continue;

    const sev = severityFromZ(z);

    const direction = z > 0 ? "spike" : "drop";
    const title = `Anomaly detected: ${cur.metric_key} ${direction}`;
    const body = `Value=${cur.value.toFixed(2)} vs seasonal median=${med.toFixed(2)} (robust z=${z.toFixed(
      2
    )}). Peer group: same month over last 3 years.`;

    // Insert insight
    await query(
      db,
      `
      INSERT INTO intel.insight
      (insight_id, insight_type, title, body, severity, status, created_at, updated_at,
       metric_key, segment_key, as_of_date, evidence_json, actionability, recency_score)
      VALUES
      (NEWID(), 'metric_anomaly', @title, @body, @severity, 'open', SYSUTCDATETIME(), SYSUTCDATETIME(),
       @metricKey, @segmentKey, CAST(GETDATE() AS date),
       @evidenceJson, @actionability, @recencyScore)
      `,
      {
        title,
        body,
        severity: sev,
        metricKey: cur.metric_key,
        segmentKey: cur.segment_key,
        evidenceJson: JSON.stringify({
          algorithm: "robust_z_median_mad_seasonal_by_month",
          current: cur.value,
          median: med,
          mad: madVal,
          z,
          peerGroupCount: values.length,
        }),
        actionability: 0.7,
        recencyScore: 1.0,
      }
    );
  }
}
```

## 4) Nightly Job: Donor Risk Scorer

### 4.1 Features
Per person:
- **Recency**: days since last gift
- **Frequency**: gifts in last 365
- **Monetary**: giving in last 365
- **Trend**: last 6 months vs previous 6 months
- **Engagement**: tags/engagement count last 365 (or other signals)
- **Capacity**: annual_capacity + giving-to-capacity ratio

### 4.2 Rules-based scoring (upgrade to ML later)
Compute `risk_score` 0–100:
- Recency points (0–40)
- Frequency points (0–15)
- Monetary points (0–15)
- Negative trend points (0–15)
- Low engagement points (0–5)
- High capacity + low giving points (0–10) (opportunity *and* “relationship risk”)

Risk levels:
- `low` < 30
- `medium` 30–59
- `high` 60–79
- `critical` >= 80

Also compute “top drivers” list.

### 4.3 Implementation

`src/jobs/donorRiskScorer.ts`
```ts
import { getDb, query } from "@/lib/db";

type Row = {
  person_id: number;
  last_gift_date: string | null;
  recency_days: number | null;
  frequency_365: number | null;
  monetary_365: number | null;
  engagement_depth_365: number | null;
  growth_trajectory_6mo_pct: number | null;
  annual_capacity: number | null;
  giving_12mo: number | null;
  capacity_gap_12mo: number | null;
  total_ltv: number | null;
};

function clamp(n: number, lo: number, hi: number) {
  return Math.max(lo, Math.min(hi, n));
}

function riskLevel(score: number): "low" | "medium" | "high" | "critical" {
  if (score >= 80) return "critical";
  if (score >= 60) return "high";
  if (score >= 30) return "medium";
  return "low";
}

export async function runDonorRiskScorer(): Promise<void> {
  const db = await getDb();

  // Pull from semantic views
  const res = await query<Row>(
    db,
    `
    SELECT
      dh.person_id,
      CONVERT(varchar(10), dh.last_gift_date, 23) AS last_gift_date,
      dh.recency_days,
      dh.frequency_365,
      dh.monetary_365,
      dh.engagement_depth_365,
      dh.growth_trajectory_6mo_pct,
      cg.annual_capacity,
      cg.giving_12mo,
      cg.capacity_gap_12mo,
      pv.total_ltv
    FROM intel.vw_donor_health dh
    LEFT JOIN intel.vw_capacity_gap cg ON cg.person_id = dh.person_id
    LEFT JOIN intel.vw_person_value pv ON pv.person_id = dh.person_id
    `
  );

  const tx = db.transaction();
  await tx.begin();

  try {
    for (const r of res.recordset) {
      const recency = r.recency_days ?? 9999;
      const freq = r.frequency_365 ?? 0;
      const mon = r.monetary_365 ?? 0;
      const eng = r.engagement_depth_365 ?? 0;
      const trend = r.growth_trajectory_6mo_pct ?? 0;
      const cap = r.annual_capacity ?? 0;
      const gap = r.capacity_gap_12mo ?? 0;
      const ltv = r.total_ltv ?? 0;

      // Points
      const recencyPts =
        recency <= 30 ? 0 :
        recency <= 90 ? 10 :
        recency <= 180 ? 20 :
        recency <= 365 ? 30 : 40;

      const freqPts =
        freq >= 12 ? 0 :
        freq >= 6 ? 5 :
        freq >= 3 ? 10 :
        freq >= 1 ? 12 : 15;

      const monPts =
        mon >= 5000 ? 0 :
        mon >= 1000 ? 5 :
        mon >= 250 ? 10 :
        mon > 0 ? 12 : 15;

      const trendPts =
        trend >= 0.1 ? 0 :
        trend >= 0 ? 5 :
        trend >= -0.25 ? 10 :
        trend >= -0.5 ? 12 : 15;

      const engPts = eng >= 10 ? 0 : eng >= 3 ? 2 : 5;

      // Capacity/gap: if high capacity but low realized giving, add risk/opportunity pressure
      const capPts =
        cap >= 25000 && gap > 20000 ? 10 :
        cap >= 10000 && gap > 8000 ? 6 :
        cap >= 5000 && gap > 4000 ? 3 : 0;

      const raw = recencyPts + freqPts + monPts + trendPts + engPts + capPts;
      const score = clamp(raw * (100 / 100), 0, 100);
      const level = riskLevel(score);

      const drivers: string[] = [];
      if (recencyPts >= 20) drivers.push(`Recency ${recency}d`);
      if (freqPts >= 10) drivers.push(`Low frequency (${freq}/yr)`);
      if (monPts >= 10) drivers.push(`Low monetary ($${mon.toFixed(0)}/yr)`);
      if (trendPts >= 10) drivers.push(`Negative trend (${(trend * 100).toFixed(0)}%)`);
      if (engPts >= 2) drivers.push(`Low engagement (${eng})`);
      if (capPts >= 6) drivers.push(`High capacity gap ($${gap.toFixed(0)})`);

      const driversJson = JSON.stringify({ drivers, components: { recencyPts, freqPts, monPts, trendPts, engPts, capPts } });

      await query(
        tx,
        `
        MERGE intel.person_score AS tgt
        USING (SELECT @personId AS person_id, 'donor_risk' AS score_type) AS src
        ON tgt.person_id = src.person_id AND tgt.score_type = src.score_type
        WHEN MATCHED THEN UPDATE SET
          score = @score,
          score_level = @level,
          drivers_json = @driversJson,
          as_of_date = CAST(GETDATE() AS date),
          updated_at = SYSUTCDATETIME()
        WHEN NOT MATCHED THEN INSERT
          (person_score_id, person_id, score_type, score, score_level, drivers_json, as_of_date, created_at, updated_at)
          VALUES (NEWID(), @personId, 'donor_risk', @score, @level, @driversJson, CAST(GETDATE() AS date), SYSUTCDATETIME(), SYSUTCDATETIME());
        `,
        { personId: r.person_id, score, level, driversJson }
      );

      // Generate insight for high-risk high-value donors
      const highValue = ltv >= 5000 || mon >= 1000 || (cap >= 25000 && (r.giving_12mo ?? 0) >= 500);
      if ((level === "high" || level === "critical") && highValue) {
        await query(
          tx,
          `
          INSERT INTO intel.insight
          (insight_id, insight_type, title, body, severity, status, created_at, updated_at,
           person_id, as_of_date, evidence_json, actionability, recency_score)
          VALUES
          (NEWID(), 'donor_risk', @title, @body, @severity, 'open', SYSUTCDATETIME(), SYSUTCDATETIME(),
           @personId, CAST(GETDATE() AS date), @evidenceJson, @actionability, @recencyScore)
          `,
          {
            personId: r.person_id,
            title: `Donor at ${level} risk`,
            body: `Risk score ${score.toFixed(0)}/100. Drivers: ${drivers.slice(0, 3).join(", ")}.`,
            severity: level === "critical" ? 0.95 : 0.75,
            evidenceJson: JSON.stringify({
              score,
              level,
              recency_days: recency,
              frequency_365: freq,
              monetary_365: mon,
              trend_6mo_pct: trend,
              engagement_depth_365: eng,
              annual_capacity: cap,
              capacity_gap_12mo: gap,
              total_ltv: ltv,
              drivers,
            }),
            actionability: 0.85,
            recencyScore: 1.0,
          }
        );
      }
    }

    await tx.commit();
  } catch (e) {
    await tx.rollback();
    throw e;
  }
```

### 5.1 Implementation

`src/jobs/opportunityMiner.ts`
```ts
import { getDb, query } from "@/lib/db";

type Opp = {
  person_id: number;
  score: number;
  title: string;
  body: string;
  evidence: any;
  severity: number;
  actionability: number;
};

export async function runOpportunityMiner(): Promise<void> {
  const db = await getDb();
  const opps: Opp[] = [];

  // A) High capacity, low giving (gap)
  {
    const res = await query<any>(
      db,
      `
      SELECT TOP 500
        cg.person_id,
        cg.annual_capacity,
        cg.giving_12mo,
        cg.capacity_gap_12mo,
        pv.total_ltv
      FROM intel.vw_capacity_gap cg
      LEFT JOIN intel.vw_person_value pv ON pv.person_id = cg.person_id
      WHERE cg.annual_capacity IS NOT NULL
        AND cg.annual_capacity >= 10000
        AND ISNULL(cg.giving_12mo,0) < 0.05 * cg.annual_capacity
      ORDER BY cg.capacity_gap_12mo DESC
      `
    );

    for (const r of res.recordset) {
      const gap = Number(r.capacity_gap_12mo ?? 0);
      const score = Math.min(100, gap / 500); // simple ranking
      opps.push({
        person_id: r.person_id,
        score,
        title: "High capacity, low realized giving",
        body: `Annual capacity ~$${Number(r.annual_capacity).toFixed(0)}; last 12mo giving $${Number(r.giving_12mo ?? 0).toFixed(0)} (gap $${gap.toFixed(0)}).`,
        evidence: r,
        severity: 0.6,
        actionability: 0.9,
      });
    }
  }

  // B) Event attendees who never donated
  {
    const res = await query<any>(
      db,
      `
      SELECT TOP 500
        t.person_id,
        COUNT(*) AS events_attended,
        MAX(t.event_date) AS last_event_date
      FROM dbo.event_ticket t
      LEFT JOIN dbo.gift g ON g.person_id=t.person_id AND g.status='posted'
      WHERE t.status IN ('attended','paid')
        AND g.person_id IS NULL
      GROUP BY t.person_id
      ORDER BY COUNT(*) DESC, MAX(t.event_date) DESC
      `
    );

    for (const r of res.recordset) {
      const score = Math.min(100, Number(r.events_attended) * 10);
      opps.push({
        person_id: r.person_id,
        score,
        title: "Event attendee who has never donated",
        body: `${r.events_attended} event(s) attended; last on ${String(r.last_event_date).slice(0, 10)}.`,
        evidence: r,
        severity: 0.35,
        actionability: 0.8,
      });
    }
  }

  // C) Commerce customers who never donated
  {
    const res = await query<any>(
      db,
      `
      SELECT TOP 500
        o.person_id,
        COUNT(*) AS orders,
        SUM(CAST(o.total AS decimal(18,2))) AS commerce_ltv,
        MAX(o.order_date) AS last_order_date
      FROM dbo.order_header o
      LEFT JOIN dbo.gift g ON g.person_id=o.person_id AND g.status='posted'
      WHERE o.status IN ('paid','fulfilled')
        AND g.person_id IS NULL
      GROUP BY o.person_id
      ORDER BY SUM(CAST(o.total AS decimal(18,2))) DESC
      `
    );

    for (const r of res.recordset) {
      const ltv = Number(r.commerce_ltv ?? 0);
      const score = Math.min(100, ltv / 50);
      opps.push({
        person_id: r.person_id,
        score,
        title: "Commerce customer who has never donated",
        body: `${r.orders} order(s), commerce LTV $${ltv.toFixed(0)}; last order ${String(r.last_order_date).slice(0, 10)}.`,
        evidence: r,
        severity: 0.4,
        actionability: 0.75,
      });
    }
  }

  // D) Subscription holders who never donated
  {
    const res = await query<any>(
      db,
      `
      SELECT TOP 500
        s.person_id,
        MIN(s.start_date) AS first_start,
        MAX(ISNULL(s.end_date, CAST(GETDATE() AS date))) AS last_seen,
        SUM(CAST(s.monthly_amount AS decimal(18,2))) AS monthly_total
      FROM dbo.subscription s
      LEFT JOIN dbo.gift g ON g.person_id=s.person_id AND g.status='posted'
      WHERE s.status IN ('active','canceled','ended')
        AND g.person_id IS NULL
      GROUP BY s.person_id
      ORDER BY SUM(CAST(s.monthly_amount AS decimal(18,2))) DESC
      `
    );

    for (const r of res.recordset) {
      const monthly = Number(r.monthly_total ?? 0);
      const score = Math.min(100, monthly * 5);
      opps.push({
        person_id: r.person_id,
        score,
        title: "Subscriber who has never donated",
        body: `Subscriptions total ~$${monthly.toFixed(0)}/mo; first start ${String(r.first_start).slice(0, 10)}.`,
        evidence: r,
        severity: 0.35,
        actionability: 0.8,
      });
    }
  }

  // E) Lost recurring donors recovery queue (sorted by annual value)
  // If you have recurring gifts table, use it. Here we approximate: donors with >=6 gifts last year and none in last 180 days.
  {
    const res = await query<any>(
      db,
      `
      WITH last_year AS (
        SELECT person_id,
               COUNT(*) AS gifts_last_year,
               SUM(CAST(amount AS decimal(18,2))) AS amt_last_year
        FROM dbo.gift
        WHERE status='posted'
          AND gift_date >= DATEFROMPARTS(YEAR(GETDATE())-1,1,1)
          AND gift_date <  DATEFROMPARTS(YEAR(GETDATE()),1,1)
        GROUP BY person_id
      ),
      recent AS (
        SELECT person_id, MAX(gift_date) AS last_gift_date
        FROM dbo.gift
        WHERE status='posted'
        GROUP BY person_id
      )
      SELECT TOP 500
        ly.person_id,
        ly.gifts_last_year,
        ly.amt_last_year,
        r.last_gift_date
      FROM last_year ly
      JOIN recent r ON r.person_id=ly.person_id
      WHERE ly.gifts_last_year >= 6
        AND DATEDIFF(day, r.last_gift_date, CAST(GETDATE() AS date)) > 180
      ORDER BY ly.amt_last_year DESC
      `
    );

    for (const r of res.recordset) {
      const annual = Number(r.amt_last_year ?? 0);
```

## 6) Insight Feed API (Next.js 16 route)

### 6.1 Ranking algorithm
`rank = severity × recency × actionability`

- `severity` in [0,1]
- `recency` decays with age (hours): `recency = exp(-hours/72)` (3-day half-ish)
- `actionability` in [0,1]

### 6.2 Route

`src/app/api/insights/feed/route.ts`
```ts
import { NextResponse } from "next/server";
import { getDb, query } from "@/lib/db";

function hoursSince(date: Date) {
  return (Date.now() - date.getTime()) / (1000 * 60 * 60);
}
function recencyScore(createdAt: Date) {
  const h = hoursSince(createdAt);
  return Math.exp(-h / 72);
}

export async function GET(req: Request) {
  const url = new URL(req.url);
  const limit = Math.min(200, Math.max(1, Number(url.searchParams.get("limit") ?? "50")));

  const db = await getDb();

  const res = await query<any>(
    db,
    `
    SELECT TOP (@limit)
      insight_id,
      insight_type,
      title,
      body,
      severity,
      status,
      created_at,
      metric_key,
      segment_key,
      person_id,
      as_of_date,
      evidence_json,
      actionability
    FROM intel.insight
    WHERE status='open'
      AND created_at >= DATEADD(day, -30, SYSUTCDATETIME())
    ORDER BY created_at DESC
    `,
    { limit }
  );

  const cards = res.recordset.map((r: any) => {
    const createdAt = new Date(r.created_at);
    const sev = Number(r.severity ?? 0);
    const act = Number(r.actionability ?? 0.5);
    const rec = recencyScore(createdAt);
    const rank = sev * rec * act;

    const evidence = (() => {
      try { return r.evidence_json ? JSON.parse(r.evidence_json) : null; } catch { return null; }
    })();

    const recommendedActions: string[] = [];
    if (r.insight_type === "donor_risk") {
      recommendedActions.push("Assign to relationship owner for outreach");
      recommendedActions.push("Review last touchpoints and recent engagement");
      recommendedActions.push("Prepare a personalized ask or care call");
    } else if (r.insight_type === "metric_anomaly") {
      recommendedActions.push("Check posting/import pipeline for delays");
      recommendedActions.push("Break down by fund/campus/channel to isolate driver");
    } else if (r.insight_type === "opportunity") {
      recommendedActions.push("Add to outreach queue with suggested next step");
      recommendedActions.push("Send tailored message based on channel history");
    }

    return {
      id: r.insight_id,
      type: r.insight_type,
      title: r.title,
      body: r.body,
      rank,
      scores: { severity: sev, recency: rec, actionability: act },
      context: {
        metricKey: r.metric_key,
        segmentKey: r.segment_key,
        personId: r.person_id,
        asOfDate: r.as_of_date,
      },
      drilldown: {
        evidence,
      },
      recommendedActions,
      createdAt: r.created_at,
    };
  });

  cards.sort((a: any, b: any) => b.rank - a.rank);

  return NextResponse.json({
    generatedAt: new Date().toISOString(),
    items: cards.slice(0, limit),
  });
}
```

### 6.3 How it powers the proactive home dashboard
- Home dashboard calls `GET /api/insights/feed?limit=50`
- Renders “Insight cards” sorted by `rank`
- Each card includes:
  - `context.personId` → link to person profile
  - `context.metricKey` → link to metric detail page
  - `drilldown.evidence` → show computed facts (no LLM needed)

## 7) Morning Briefing Generator (LLM, evidence-grounded)

### 7.1 Grounding rule
The LLM only receives:
- Top ranked insights (title/body + evidence JSON)
- Metric deltas from `intel.metric_snapshot` (already computed)
No raw DB access, no free-form numbers.

### 7.2 Implementation

`src/jobs/morningBriefing.ts`
```ts
import { getDb, query } from "@/lib/db";
import OpenAI from "openai";

type BriefInput = {
  date: string;
  topInsights: any[];
  metricDeltas: any[];
};

export async function generateMorningBriefing(): Promise<{ narrative: string; payload: BriefInput }> {
  const db = await getDb();

  const insightsRes = await query<any>(
    db,
    `
    SELECT TOP 20
      insight_id, insight_type, title, body, severity, created_at, metric_key, segment_key, person_id, evidence_json, actionability
    FROM intel.insight
    WHERE status='open'
      AND created_at >= DATEADD(hour, -24, SYSUTCDATETIME())
    ORDER BY (severity * ISNULL(actionability,0.5)) DESC, created_at DESC
    `
  );

  const metricsRes = await query<any>(
    db,
    `
    SELECT TOP 30
      metric_key, segment_key, as_of_date, value, wow_delta, mom_delta, yoy_delta
    FROM intel.metric_snapshot
    WHERE as_of_date = CAST(GETDATE() AS date)
    ORDER BY ABS(ISNULL(yoy_delta,0)) DESC, ABS(ISNULL(mom_delta,0)) DESC
    `
  );

  const topInsights = insightsRes.recordset.map((r) => ({
    id: r.insight_id,
    type: r.insight_type,
    title: r.title,
    body: r.body,
    severity: Number(r.severity ?? 0),
    actionability: Number(r.actionability ?? 0.5),
    created_at: r.created_at,
    context: { metric_key: r.metric_key, segment_key: r.segment_key, person_id: r.person_id },
    evidence: (() => { try { return r.evidence_json ? JSON.parse(r.evidence_json) : null; } catch { return null; } })(),
  }));

  const metricDeltas = metricsRes.recordset.map((r) => ({
    metric_key: r.metric_key,
    segment_key: r.segment_key,
    value: Number(r.value),
    wow_delta: r.wow_delta === null ? null : Number(r.wow_delta),
    mom_delta: r.mom_delta === null ? null : Number(r.mom_delta),
    yoy_delta: r.yoy_delta === null ? null : Number(r.yoy_delta),
  }));

  const payload: BriefInput = {
    date: new Date().toISOString().slice(0, 10),
    topInsights,
    metricDeltas,
  };

  const apiKey = process.env.OPENAI_API_KEY;
  if (!apiKey) {
    return {
      narrative:
        "Morning briefing unavailable (missing OPENAI_API_KEY). See insights feed for details.",
      payload,
    };
  }

  const client = new OpenAI({ apiKey });

  const system = `
You are Sozo Morning Briefing.
Rules:
- Use ONLY the facts provided in the JSON input.
- Do NOT invent numbers, names, or causes.
- If data is insufficient, say so explicitly.
- Output: 6-10 bullet points + a short 'Recommended Focus Today' section.
- Keep it concise and operational.
`;

  const user = `
Generate a morning briefing for ministry leadership based on this computed data.
JSON:
${JSON.stringify(payload, null, 2)}
`;

  const resp = await client.chat.completions.create({
    model: "gpt-4.1-mini",
    temperature: 0.2,
    messages: [
      { role: "system", content: system },
      { role: "user", content: user },
    ],
  });

  const narrative = resp.choices[0]?.message?.content?.trim() ?? "No briefing generated.";

  // Optionally store as an insight card for dashboard
  await query(
    db,
    `
    INSERT INTO intel.insight
    (insight_id, insight_type, title, body, severity, status, created_at, updated_at,
     as_of_date, evidence_json, actionability, recency_score)
    VALUES
    (NEWID(), 'morning_briefing', 'Morning Briefing', @body, 0.5, 'open',
     SYSUTCDATETIME(), SYSUTCDATETIME(), CAST(GETDATE() AS date), @evidenceJson, 0.6, 1.0)
    `,
    { body: narrative, evidenceJson: JSON.stringify(payload) }
  );

  return { narrative, payload };
}
```
